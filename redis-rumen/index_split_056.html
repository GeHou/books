* Poly　　　　　: 1021 (That is actually x^16 + x^12 + x^5 + 1)

* Initialization　　　: 0000

* Reflect Input byte　　: False

* Reflect Output CRC　　: False

*/

uint16_t crc16(const char *buf, int len) {

int counter;
uint16_t crc = 0;
for (counter = 0; counter < len; counter++)
crc = (crc<<8) ^ crc16tab[((crc>>8) ^ *buf++)&0x00FF];
return crc;

}
